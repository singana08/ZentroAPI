TESTING ENDPOINTS WITH CURL
================================================================================

# Get JWT Token (from /api/auth/verify-otp endpoint)
TOKEN="your-jwt-token-here"

# Create Service Request
curl -X POST http://localhost:5000/api/service-request \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "bookingType": "book_now",
    "mainCategory": "Cleaning",
    "subCategory": "Deep Cleaning",
    "date": "2024-11-01",
    "location": "123 Main St"
  }'

# Get User Service Requests
curl http://localhost:5000/api/service-request/user/{userId} \
  -H "Authorization: Bearer $TOKEN"

# Get Pending Requests Only
curl "http://localhost:5000/api/service-request/user/{userId}?status=Pending" \
  -H "Authorization: Bearer $TOKEN"

# Update Service Request
curl -X PUT http://localhost:5000/api/service-request/{id} \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"bookingType": "schedule_later", "mainCategory": "Cleaning", "subCategory": "Deep Cleaning", "date": "2024-11-02", "location": "456 Oak Ave"}'

# Cancel Service Request
curl -X DELETE http://localhost:5000/api/service-request/{id}/cancel \
  -H "Authorization: Bearer $TOKEN"
